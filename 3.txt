
import pandas as pd

#Load dataset

df= pd.read_csv("iris_dataset.csv")

#Create categorical groups based on sepal length

df['sepal_group'] = pd.cut(df['sepal_length'], bins=3, labels=["Low", "Medium", "High"])

#Group by categorical variable and calculate statistics

grouped = df.groupby('sepal_group')['petal_length']

print("Mean:\n", grouped.mean())

print("\nMedian:\n", grouped.median())

print("\nMinimum:\n", grouped.min())

print("\nMaximum:\n", grouped.max())

print("\nStandard Deviation:\n", grouped.std())

# Create numeric lists for each category
low_list = df[df['sepal_group']=="Low"]['petal_length'].tolist()
print("\nLow Group List:\n", low_list)

import pandas as pd

df = pd.read_csv("iris.csv.txt")

grouped = df.groupby('species')

for name, group in grouped:

    print("\n----- Species:", name, "-----")

    print("\nMean:")
    print(group.mean(numeric_only=True))

    print("\nStandard Deviation:")
    print(group.std(numeric_only=True))

    print("\nPercentiles:")
    print(group.describe())
