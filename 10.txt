import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Load dataset
df = sns.load_dataset("iris")

print("========== DATA PREVIEW ==========")

print(df.head())

print("\n========== FEATURES & DATA TYPES ==========")

print(df.dtypes)

# HISTOGRAMS
print("\n========== HISTOGRAMS ==========")

df.hist(figsize=(10, 8))

plt.suptitle("Histograms of Iris Features")

plt.show()

# BOXPLOTS
print("\n========== BOXPLOTS ==========")

features = df.columns[:-1]   # exclude species column

for col in features:

    plt.figure()

    sns.boxplot(x="species", y=col, data=df)

    plt.title(f"Boxplot of {col}")

    plt.show()

# BASIC OBSERVATIONS
print("\n========== OBSERVATIONS ==========")

print("1. Sepal features show overlap between species.")

print("2. Petal features clearly separate species.")

print("3. Some outliers are present in sepal_width.")

print("4. Petal length and width are most important features.")
