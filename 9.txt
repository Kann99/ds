import matplotlib.pyplot as plt

#

import seaborn as sns



df = sns.load_dataset("titanic")

print("========== DATA PREVIEW ==========")

print(df.head())

print("\n========== DATA INFO ==========")

print(df.info())

#


print("\n========== BOXPLOT ==========")

sns.boxplot(x="sex", y="age", hue="survived", data=df)

plt.title("Age Distribution by Gender and Survival")

plt.xlabel("Gender")

plt.ylabel("Age")

plt.show()

#



print("\n========== OBSERVATIONS ==========")

print("1. Females have higher survival rate than males.")

print("2. Children (lower age) have higher survival chances.")

print("3. Male age distribution is wider than females.")

print("4. Older passengers have lower survival rate.")

print("5. Some outliers are present in age.")
